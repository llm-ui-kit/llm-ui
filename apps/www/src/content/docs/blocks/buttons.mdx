# Installation

<PackageInstall client:load packages={["llm-ui", "@llm-ui/buttons"]} />

# Quick start

# Prompt

<CodeBlock
  className="whitespace-pre-wrap"
  codeToHtmlProps={{
    lang: "typescript",
  }}
  code={`import { buttonPrompt } from "@llm-ui/buttons";\n\nconsole.log(buttonPrompt);\n// => "${buttonPrompt.slice(0, 53)}..."`}
/>
